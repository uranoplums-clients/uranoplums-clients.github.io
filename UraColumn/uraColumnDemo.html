Sample 2 - Add new columns -

New columns add before, before first, after or after last of target DIV.

#target10

The Base Div

Codes

First New source

var s2Idx1 = 2;
$('#s2First').on('click',
  function(){
  if (s2Idx1 > 5)
    s2Idx1 = 2;

  $('#target10').uraColumn({
    append: {
      location: 'first',
      width: 80,
      className: 'demoTarget' +
        (s2Idx1++) + ' s2New',
      text: 'First'
    }
  });
});

Options

$(selector).uraColumn({ // selector is target to append it.
    append: { // for append column
      object: null, // if append target object then setting selector or jQuery object.
                    // if null this option then make new element.
      location: 'after', // choose append to.
      float: 'left', // setting css style float
      width: null,  // setting css style width
      padding: null, // setting css style padding
      margin: null,  // setting css style margin
      text: null,  // setting css style text
      idName: null,  // setting css style id
      className: 'uraColumn-inner'  // setting css style class
    },
    remove: {
      selector: null, // remove target
      location: 'after', // choose append to, if 'to' setting.
      to: null //
    },
    outer: {
      idName: null,
      className: 'uraColumn-outer'
    },
    resize: {
      sync: true,
      beforeFunc: function($self, $parent){},
      afterFunc: function($self, $parent){}
    },
    self: {
      float: 'left'
    }
);
